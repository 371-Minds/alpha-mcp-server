### **Frappe/ERPNext Integration**

```
# Agent Wallet Frappe Integration (Pseudocode)
class AgentWalletIntegration:
    """
    Frappe hooks for agent wallet management
    """
    
    def on_compute_allocation(self, agent_id, requested_specs):
        # Check wallet balance before provisioning
        wallet = get_agent_wallet(agent_id)
        estimated_cost = calculate_compute_cost(requested_specs)
        
        if wallet.available >= estimated_cost:
            # Approve and deduct
            provision_compute(agent_id, requested_specs)
            wallet.deduct(estimated_cost)
            log_transaction(agent_id, "APPROVED", estimated_cost)
            return {"status": "approved", "new_balance": wallet.available}
        else:
            # Reject with suggestion
            affordable_specs = get_affordable_alternatives(wallet.available)
            log_transaction(agent_id, "REJECTED", estimated_cost)
            return {
                "status": "rejected",
                "reason": "insufficient_balance",
                "suggestion": affordable_specs
            }
    
    def on_billing_cycle(self):
        # Monthly reconciliation
        all_wallets = get_all_wallets()
        for wallet in all_wallets:
            if wallet.spent > wallet.allocated:
                # Flag for review — potential runaway agent
                create_issue(wallet.agent_id, "OVERSPEND")
        
        # Auto-refill on 15th
        for wallet in all_wallets:
            if wallet.days_remaining >= 15:
                wallet.refill(allocated_amount)
```
